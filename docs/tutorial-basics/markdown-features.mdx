# React

export const Button = ({ children, color }) => (
  <button className={"btn-" + color}>{children}</button>
);

export const Card = ({ children, title, imgPath }) => (
  <div className={"card"}>
    <div class="card-title">{title}</div>
    <div class="card-body">
      <img src={imgPath} />
      {children}
    </div>
  </div>
);

```jsx
<Card title="title" imgPath="https://picsum.photos/740/500">
  <p>Lorem ipsum dolor sit amet, consectetur adipiscing elit. Nullam euismod</p>
  <Button color="primary">read more</Button>
  <button className="btn">save for later</button>
</Card>
```
